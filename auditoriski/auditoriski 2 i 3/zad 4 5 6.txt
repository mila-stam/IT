    protected void dodadi_click(object sender, EventArgs e)
    {
        status.Text = " ";
        ListItem item = new ListItem(imeValuta.Text, vrednostValuta.Text);
        listaValuti.Items.Add(item);

        imeValuta.Text = " ";
        vrednostValuta.Text = " ";

        total.Text = listaValuti.Items.Count.ToString();
    }

    protected void izbrishi_click(object sender, EventArgs e)
    {
        status.Text = " ";
        if (listaValuti.SelectedIndex != -1)
        {
            listaValuti.Items.Remove(listaValuti.SelectedItem);
            total.Text = listaValuti.Items.Count.ToString();
        }
        else
        {
            status.Text = "Нема селектирана валута";
        }
    }

    protected void listaValuti_indexChange(object sender, EventArgs e)
    {
        int value = Convert.ToInt32(listaValuti.SelectedValue);

        konvertirana.Text = Convert.ToString(Convert.ToInt32(konvertVrednost.Text) * value);
    }
}
