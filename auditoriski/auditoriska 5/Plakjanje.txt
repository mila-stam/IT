namespace WebApplication1
{
    public partial class _Default : Page
    {
        protected void Page_Load(object sender, EventArgs e)
        {
            korpa.DataTextField = "Text";
            korpa.DataValueField = "Value";
            korpa.DataSource = (ArrayList)Session["cart"];
            korpa.DataBind();
            int totalValue = 0;

            foreach(ListItem i in korpa.Items)
            {
                totalValue =+ Convert.ToInt32(i.Value);
            }

            kupi.Text = "Kupi ($ " + totalValue.ToString() + ")";  
        }

        protected void back_Click(object sender, EventArgs e)
        {
            Response.Redirect("Katalog.aspx");
        }

        protected void kupi_Click(object sender, EventArgs e)
        {

        }
    }
}
