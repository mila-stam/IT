    public partial class Glasaj : System.Web.UI.Page
    {
        protected void Page_Load(object sender, EventArgs e)
        {
            if (!IsPostBack)
            {
                lbPredmeti.Items.Add(new ListItem("Internet tehnologii", "Goce"));
                lbPredmeti.Items.Add(new ListItem("Internet", "Goce1"));
                lbPredmeti.Items.Add(new ListItem("Digitalna elektornika", "Goce2"));

                lbKrediti.Items.Add("6");
                lbKrediti.Items.Add("5.5");
                lbKrediti.Items.Add("4");
            }

        }

        protected void lbPredmeti_selected(object sender, EventArgs e)
        {
            int selectedIndex = lbPredmeti.SelectedIndex;
            if(selectedIndex >= 0 && selectedIndex < lbPredmeti.Items.Count)
            {
                lbKrediti.SelectedIndex = selectedIndex;
            }
            else
            {
                lbKrediti.SelectedIndex = -1;
            }
            lblProfesor.Text = lbPredmeti.SelectedValue;
        }

        protected void dodadi_click(object sender, EventArgs e)
        {
            var subject = predmet.Text;
            var credits = krediti.Text;

            lbPredmeti.Items.Add(subject);
            lbKrediti.Items.Add(credits);
        }

        protected void izbrishi_click(object sender, EventArgs e)
        {
            int selectedIndex = lbPredmeti.SelectedIndex;

            if (selectedIndex != -1 && selectedIndex < lbKrediti.Items.Count) {
                lbPredmeti.Items.RemoveAt(selectedIndex);
                lbKrediti.Items.RemoveAt(selectedIndex);

                lbPredmeti.SelectedIndex = -1;
                lblProfesor.Text = " ";
            }
        }
    }
}
